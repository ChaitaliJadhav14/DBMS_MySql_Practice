Task7: create new database with name projectDB and create table with column and with required constraint and insert data in it for table Company, Bank, Hospital, Hotel, Restaurant.

create database projectDB;
USE projectDB;
CREATE TABLE Company (
     Company_id INT PRIMARY KEY AUTO_INCREMENT,
     name VARCHAR(100) NOT NULL UNIQUE,
     industry VARCHAR(50) NOT NULL,
     founded_year INT CHECK (founded_year >= 1800),
     headquarters VARCHAR(100) NOT NULL
);

CREATE TABLE Hospital (
     Hospital_id INT  PRIMARY KEY AUTO_INCREMENT,
     name VARCHAR(100) NOT Null UNIQUE,
     location VARCHAR(100) NOT Null,
     capacity INT CHECK(capacity > 0),
     established_year INT CHECK (established_year >= 1800)
    );

CREATE TABLE Hotel(
     hotel_id INT PRIMARY KEY AUTO_INCREMENT,
     name VARCHAR(100) NOT NULL UNIQUE,
     location VARCHAR(100) NOT NULL,
     rating DECIMAL(2,1) CHECK (rating >= 0 AND rating <=5),
     established_year INT CHECK (established_year >=1800)
     
    );

select * from Hotel;
CREATE TABLE Restaurant (
    restaurant_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL UNIQUE,
    cuisine VARCHAR(50) NOT NULL,
    location VARCHAR(100) NOT NULL,
    rating DECIMAL(2,1) CHECK (rating >= 0 AND rating <= 5),
    hotel_id INT,
    FOREIGN KEY (hotel_id) REFERENCES Hotel(hotel_id) ON DELETE SET NULL 
);
