Task 35:Write a stored procedure using a REPEAT loop to print the first 10 terms of the Fibonacci series.

use ycpdb;
delimiter $$
create procedure Fibonacci_series()
begin
declare a int default 0;
declare b int default 1;
declare c int;
declare i int default 1;
declare result VARCHAR(255) DEFAULT "";
lb1:REPEAT
SET result = concat(result," ",a);
SET c = a + b;
SET a = b;
SET b = c;
SET i = i +1;
UNTIL i > 10 
END REPEAT lb1;
select result AS fibonacciseries;
end $$
delimiter ;
drop procedure Fibonacci_series;
call Fibonacci_series();
