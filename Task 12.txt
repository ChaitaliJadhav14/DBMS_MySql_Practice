2.	Write a query to display all students and their institute names. If a student is not enrolled in any institute, display "Not Assigned" in place of institute name.
3.	Write a query to display all institutes along with the students enrolled in them. If an institute has no students, still display the institute with "No Student".
4.	Write a query to display all students and all institutes, even if there is no matching data between them.

CREATE TABLE Institute (
	institute_id INT PRIMARY KEY AUTO_INCREMENT,
        institute_name VARCHAR(100) NOT NULL,
        city VARCHAR(50) NOT NULL
);
create table Students (
     student_id INT PRIMARY KEY AUTO_INCREMENT,
     student_name VARCHAR(100) NOT NULL,
     age INT CHECK(age >=15),  
     institute_id INT,
     course VARCHAR(100) NOT NULL,
     CONSTRAINT fk_institute FOREIGN KEY (institute_id) REFERENCES Institute(institute_id)
         ON DELETE SET NULL ON UPDATE CASCADE
     );
