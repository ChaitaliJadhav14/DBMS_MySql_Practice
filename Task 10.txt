Tasl10: Task on join for Restaurant and Order table, Restaurant (restaurant_id, name ,location), Orders ( order_id ,restaurant_id ,customer_name , amount DECIMAL(8,2), order_date DATE,FOREIGN KEY (restaurant_id) REFERENCES Restaurant);
1.	List all orders with restaurant names
2.	Show restaurant names with their order amounts
3.	Find all orders placed at restaurants in ‘Mumbai’
4.	Get total sales (sum of amount) for each restaurant
5.	Find restaurants that have no orders.
6.	List customers and restaurant names.
7.	Show restaurant and order details.

CREATE TABLE Orders (order_id INT, restaurant_id INT, customer_name VARCHAR(100), amount DECIMAL(8,2),
 order_date DATE, FOREIGN KEY (restaurant_id) REFERENCES Restaurant(restaurant_id));

SELECT o.order_id, o.customer_name, o.amount, o.order_date, r.name AS restaurant_name
FROM Orders o
JOIN Restaurant r ON o.restaurant_id = r.restaurant_id;

SELECT r.name AS restaurant_name, o.amount
FROM Orders o
JOIN Restaurant r ON o.restaurant_id = r.restaurant_id;

SELECT o.order_id, o.customer_name, o.amount, o.order_date, r.name AS restaurant_name
FROM Orders o
JOIN Restaurant r ON o.restaurant_id = r.restaurant_id
WHERE r.location = 'Mumbai';
        
SELECT r.name AS restaurant_name, SUM(o.amount) AS total_sales
FROM Orders o
JOIN Restaurant r ON o.restaurant_id = r.restaurant_id
GROUP BY r.name;

SELECT r.restaurant_id, r.name, r.location
FROM Restaurant r
LEFT JOIN Orders o ON r.restaurant_id = o.restaurant_id
WHERE o.order_id IS NULL;

SELECT o.customer_name, r.name AS restaurant_name
FROM Orders o
JOIN Restaurant r ON o.restaurant_id = r.restaurant_id;

SELECT r.restaurant_id, r.name AS restaurant_name, r.location,
       o.order_id, o.customer_name, o.amount, o.order_date
