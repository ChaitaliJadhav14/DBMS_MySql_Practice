Task27: while Loop to print addition number from 1 to 10.

use ycpdb;
delimiter $$
create procedure calculation(IN p_limit int)
begin
declare i int default 1;
declare total_sum int default 0;
lb1: WHILE i <= p_limit DO
SET total_sum = total_sum +i;
SET i = i + 1;
END WHILE lb1;
select total_sum AS calculatedsum;
end $$
delimiter ;
call calculation(10);
